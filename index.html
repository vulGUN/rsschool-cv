function tribonacci(signature, n) {
  let arr = [...signature];
  if (n === 0) return [];
  else if (n === 1) return [arr[0]];
  else if (n === 2) return [arr[0], arr[1]];
  for (let i = 0; i < n - 3; i++) {
      arr.push(arr[i] + arr[i + 1] + arr[i + 2]);
  }
  return arr;
}
